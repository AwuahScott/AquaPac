📊 Google Sheets Sync
☁️ Auto-syncs on every login — phone, laptop, boss all see the same data. Records push to YOUR Google Sheet instantly when saved.

Setup (one time):

  1. Open Google Sheets new blank sheet
  2. Click Extensions → Apps Script
  3. Delete all text, paste the script below
  4. Click Deploy → New deployment → Web app
  5. Set "Who has access" to Anyone → Deploy
  6. Copy the Web App URL and paste it below
// ── AquaPac Apps Script ── // Paste this ENTIRE script. Deploy as Web App, access: Anyone. // IMPORTANT: Redeploy as NEW VERSION after replacing old script. function doPost(e) { try { var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); var data = JSON.parse(e.postData.contents); var headers = [ 'date','salesperson','taken','takings','produced','sold', 'prevLeft','leakage','bossWater','price','workerRate','prodCost', 'gross','totalExpenses','net','prevDayNet','settlement', 'totalWater','leftover','expenses','savedAt' ]; // Write header row if sheet is empty if (sheet.getLastRow() === 0) { sheet.appendRow([ 'Date','Salesperson','Bags Taken','Batches','Produced', 'Sold at Site','Prev Leftover','Leakage','Drawings (Owner)', 'Price/Bag','Worker Rate','Prod Cost','Gross (GHC)', 'Expenses (GHC)','Net (GHC)','Prev Day Net', 'Settlement (Boss)','Total Water','Leftover', 'Expense Details','Saved At' ]); } // Find and DELETE existing row for this date (prevents duplicates) var allRows = sheet.getDataRange().getValues(); for (var i = 1; i < allRows.length; i++) { if (String(allRows[i][0]) === String(data.date)) { sheet.deleteRow(i + 1); break; } } // Append updated row sheet.appendRow([ data.date, data.salesperson, data.taken, (data.takings || []).join('+'), data.produced, data.sold, data.prevLeft, data.leakage, data.bossWater || 0, data.price, data.workerRate || 0.25, data.prodCost || 0, data.gross, data.totalExpenses, data.net, data.prevDayNet != null ? data.prevDayNet : 'N/A', data.settlement != null ? data.settlement : 'Carried forward', data.totalWater, data.leftover, JSON.stringify(data.expenses || []), data.savedAt ]); return ContentService .createTextOutput(JSON.stringify({ status: 'ok' })) .setMimeType(ContentService.MimeType.JSON); } catch (err) { return ContentService .createTextOutput(JSON.stringify({ status: 'error', msg: err.toString() })) .setMimeType(ContentService.MimeType.JSON); } } function doGet(e) { try { var action = (e && e.parameter && e.parameter.action) ? e.parameter.action : ''; var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); var n = sheet.getLastRow() - 1; // Return all records as JSON so the app can pull them on login if (action === 'getAll' && n > 0) { var rows = sheet.getDataRange().getValues(); var records = []; for (var i = 1; i < rows.length; i++) { var r = rows[i]; if (!r[0]) continue; var expRaw = r[19]; var expenses = []; try { expenses = JSON.parse(expRaw); } catch(x) { expenses = []; } var takingsRaw = String(r[3] || ''); var takings = takingsRaw ? takingsRaw.split('+').map(Number).filter(Boolean) : []; records.push({ date: String(r[0]), salesperson: String(r[1]), taken: Number(r[2]) || 0, takings: takings, produced: Number(r[4]) || 0, sold: Number(r[5]) || 0, prevLeft: Number(r[6]) || 0, leakage: Number(r[7]) || 0, bossWater: Number(r[8]) || 0, price: Number(r[9]) || 5, workerRate: Number(r[10]) || 0.25, prodCost: Number(r[11]) || 0, gross: Number(r[12]) || 0, totalExpenses: Number(r[13]) || 0, net: Number(r[14]) || 0, prevDayNet: r[15] === 'N/A' ? null : (Number(r[15]) || null), settlement: r[16] === 'Carried forward' ? null : (Number(r[16]) || null), totalWater: Number(r[17]) || 0, leftover: Number(r[18]) || 0, expenses: expenses, prodSalesAmt: (Number(r[5])||0) * (Number(r[9])||5), savedAt: String(r[20] || new Date().toISOString()) }); } return ContentService .createTextOutput(JSON.stringify({ status: 'ok', records: records })) .setMimeType(ContentService.MimeType.JSON); } // Default: just return row count return ContentService .createTextOutput(JSON.stringify({ status: 'ok', rows: n })) .setMimeType(ContentService.MimeType.JSON); } catch (err) { return ContentService .createTextOutput(JSON.stringify({ status: 'error', msg: err.toString() })) .setMimeType(ContentService.MimeType.JSON); } }
